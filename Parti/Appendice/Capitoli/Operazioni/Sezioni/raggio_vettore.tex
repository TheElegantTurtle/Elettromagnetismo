\begin{equation}
  \label{app:eqn:grad_r}
    \frac{\vb{r}-\vb{r}'}{\abs{\vb{r}-\vb{r}'}^3}=-\grad[\rec{\abs{\vb{r}-\vb{r'}}}]=\grad'[\rec{\abs{\vb{r}-\vb{r'}}}]
\end{equation}

  \label{app:eqn:laplacian_r}
    \laplacian{\vb{r}-\vb{r}'}=4\pi\delta^3(\vb{r}-\vb{r}')
\end{equation}
